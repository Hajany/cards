# This enables LDAP authentication, but does not include any specific bind information.
[feature name=ldap-authentication]

[variables]
    oak.version=1.8.8

[artifacts startLevel=15]
    org.apache.jackrabbit/oak-auth-external/${oak.version}
    org.apache.jackrabbit/oak-auth-ldap/${oak.version}

# Configuration for authentication and authorization
[configurations]
  org.apache.jackrabbit.oak.security.authentication.ldap.impl.LdapIdentityProvider
    adminPool.lookupOnValidate=B"true"
    adminPool.maxActive=L"8"
    userPool.lookupOnValidate=B"true"
    userPool.maxActive=L"8"

  org.apache.jackrabbit.oak.spi.security.authentication.external.impl.ExternalLoginModuleFactory
    idp.name="ldap"
    jaas.controlFlag="SUFFICIENT"
    jaas.ranking=I"50"
    jaas.realmName=""
    sync.handlerName="default"

  org.apache.jackrabbit.oak.spi.security.authentication.external.impl.DefaultSyncHandler
    enableRFC7613UsercaseMappedProfile=B"false"
    handler.name="default"
    user.pathPrefix="/ldapUsers"
    user.propertyMapping=["rep:fullname\=cn"]
